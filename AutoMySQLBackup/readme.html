فایل "automysqlbackup" را به مسیر ‎/usr/local/bin/automysqlbackup منتقل کنید و پرمیژن فایل را تغییر دهید:

chmod +x /usr/local/bin/automysqlbackup

فایل "automysqlbackup.conf" را به مسیر ‎/etc/automysqlbackup/automysqlbackup.conf منتقل کنید. تنظیمات مربوط به بک آپ گیری از دیتابیس را باید در این فایل اعمال کنید:


CONFIG_mysql_dump_username='backupuser'‎
CONFIG_mysql_dump_password='password'‎
CONFIG_mysql_dump_host='127.0.0.1'‎
CONFIG_backup_dir='/var/backup/db'‎
‎#CONFIG_db_exclude=('information_schema' 'mysql' 'performance_schema')‎
CONFIG_mysql_dump_port=3306‎
CONFIG_mysql_dump_commcomp='no'‎
CONFIG_mysql_dump_usessl='no'‎
CONFIG_mysql_dump_max_allowed_packet='16MB'‎
CONFIG_mysql_dump_single_transaction='no'‎
CONFIG_mysql_dump_full_schema='yes'‎
CONFIG_mysql_dump_dbstatus='yes'‎
CONFIG_mysql_dump_create_database='yes'‎
CONFIG_mysql_dump_use_separate_dirs='yes'‎
CONFIG_mysql_dump_compression='gzip'‎
CONFIG_mysql_dump_latest='yes'‎
CONFIG_mysql_dump_latest_clean_filenames='yes'‎
CONFIG_mysql_dump_differential='no'‎
CONFIG_encrypt='no'‎
umask 0077‎

فایل "autobackup-mysql" را به مسیر ‎/etc/cron.d/autobackup-mysql انتقال دهید.
